        // Main Core Syringe Pump Math Converter

        function recalculatePumpRate() {

            const drug = DRUG_DATABASE.find(d => d.id === selectedDrugId);

            if (!drug || !drug.pumpPresets) return;


            const weight = parseFloat(document.getElementById('globalWeight').value) || 60;

            const amount = parseFloat(document.getElementById('pumpAmountInput').value) || 0;

            const volume = parseFloat(document.getElementById('pumpVolumeInput').value) || 50;

            const targetDose = parseFloat(document.getElementById('pumpDoseInput').value) || 0;


            if (amount <= 0 || volume <= 0 || targetDose <= 0) {

                document.getElementById('pumpResultValue').innerText = "0.0";

                document.getElementById('pumpConcLabel').innerText = "Chưa xác định";

                return;

            }


            const currentPreset = drug.pumpPresets[0]; // Fetch meta-units

            const amountUnit = currentPreset.amountUnit; // "mg" or "mcg"

            const doseUnit = currentPreset.doseUnit;     // "mcg/kg/phút", "mg/kg/giờ", etc.


            // 1. Calculate Concentration in both units

            let concMgMl = 0;

            let concMcgMl = 0;


            if (amountUnit === 'mg') {

                concMgMl = amount / volume;

                concMcgMl = (amount * 1000) / volume;

            } else { // "mcg" (e.g. Dexmedetomidine)

                concMgMl = (amount / 1000) / volume;

                concMcgMl = amount / volume;

            }

        // Beautiful Toast message implementation

        function showToast(message, type = "success") {

            const container = document.getElementById('toastContainer');

            if (!container) return;

            const toast = document.createElement('div');

            

            let bgClass = "bg-slate-900 text-white";

            let icon = "info";

            if (type === "success") {

                bgClass = "bg-emerald-600 text-white";

                icon = "check-circle";

            } else if (type === "error") {

                bgClass = "bg-rose-600 text-white";

                icon = "alert-triangle";

            }


            toast.className = `${bgClass} flex items-center gap-2.5 px-4 py-3 rounded-xl shadow-lg text-xs font-semibold tracking-wide transition-all duration-300 transform translate-y-2 opacity-0`;

            toast.innerHTML = `

                <i data-lucide="${icon}" class="w-4 h-4"></i>

                <span>${message}</span>

            `;

            

            container.appendChild(toast);

            lucide.createIcons();


            // Slide up & in

            setTimeout(() => {

                toast.classList.remove('translate-y-2', 'opacity-0');

            }, 50);


            // Auto dismiss

            setTimeout(() => {

                toast.classList.add('translate-y-2', 'opacity-0');

                setTimeout(() => {

                    toast.remove();

                }, 300);

            }, 3000);

        }
